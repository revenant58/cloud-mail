Auth System

Authentication

Semua request ke /api/v1/* memerlukan header API Key:

x-api-key: cm_your-api-key-here

Dapatkan API Key dari: Admin Panel → Management → API Keys → Generate Key

⚠️ Peringatan API Key hanya ditampilkan SEKALI saat pembuatan. Copy dan simpan di tempat aman. Key tidak bisa di-retrieve lagi dari database.
ℹ️ Autentikasi Flow x-api-key → SHA-256 hash → DB lookup → Status check → Scope check → Endpoint access

Scopes

Setiap API Key memiliki daftar scope yang menentukan endpoint mana yang bisa diakses:

ScopeDescriptionEndpoints
users Create, list, delete users /v1/users
emails List, read, delete emails /v1/emails
stats View system statistics /v1/stats
🔑 Contoh Jika key hanya memiliki scope ["emails"], request ke /v1/users akan return 403 Forbidden.

Users

GET /api/v1/users scope: users

Daftar semua user dengan pagination.

ParamTypeDefaultDescription
numint1Halaman
sizeint20Item per halaman (max 50)
typeint—Filter tipe user

Response:

{
  "code": 200,
  "data": {
    "list": [...],
    "total": 42
  }
}
POST /api/v1/users scope: users

Buat user baru.

Body (JSON):

{
  "email": "user@example.com",
  "password": "your-password"
}

Response:

{
  "code": 200
}
DELETE /api/v1/users scope: users

Hapus user secara permanen (physical delete).

Body (JSON):

{
  "userId": 123
}
⚠️ Hati-hati Operasi ini permanent. User yang dihapus tidak bisa dikembalikan.

Emails

GET /api/v1/emails scope: emails

Daftar email dengan filter dan pagination.

ParamTypeDefaultDescription
numint1Halaman
sizeint20Item per halaman (max 50)
typeint—0 = inbox, 1 = sent, 2 = trash

Response:

{
  "code": 200,
  "data": {
    "list": [
      {
        "emailId": 1,
        "from": "sender@example.com",
        "to": "user@revmail.my.id",
        "subject": "Hello",
        "body": "...",
        "type": 0,
        "createTime": "2026-07-20T12:00:00.000Z"
      }
    ],
    "total": 150
  }
}
GET /api/v1/emails/:emailId scope: emails

Detail email berdasarkan ID.

Contoh:

// GET /api/v1/emails/42

Response (success):

{
  "code": 200,
  "data": {
    "emailId": 42,
    "from": "sender@example.com",
    "to": "user@revmail.my.id",
    "subject": "Hello",
    "body": "Full email content...",
    "type": 0,
    "createTime": "2026-07-20T12:00:00.000Z"
  }
}

Response (not found):

{
  "code": 404,
  "message": "Email not found"
}
DELETE /api/v1/emails scope: emails

Soft delete email (pindah ke trash). Kirim comma-separated IDs.

Body (JSON):

{
  "emailIds": "1,2,3"
}

Response:

{
  "code": 200
}

Error Codes

CodeMeaningCause
400 Bad Request Missing required parameters
401 Unauthorized Invalid / missing / expired API key
403 Forbidden Key valid tapi scope tidak cukup
404 Not Found Resource tidak ditemukan

Contoh Error Response:

// Missing API key
{
  "code": 401,
  "message": "Authentication has expired. Please sign in again"
}

// Invalid key
{
  "code": 401,
  "message": "Invalid API key"
}

// Missing scope
{
  "code": 403,
  "message": "Missing required scope: emails"
}

// Key disabled
{
  "code": 401,
  "message": "API key is disabled"
}

// Key expired
{
  "code": 401,
  "message": "API key has expired"
}

Discord.js Integration

Setup

// config.js
module.exports = {
  CLOUDMAIL_BASE: 'https://mail.revmail.my.id/api',
  CLOUDMAIL_API_KEY: process.env.CLOUDMAIL_API_KEY,
};

// api.js — CloudMail API Client
const fetch = require('node-fetch');
const { CLOUDMAIL_BASE, CLOUDMAIL_API_KEY } = require('./config');

const headers = { 'x-api-key': CLOUDMAIL_API_KEY };

async function listEmails(page = 1, type) {
  let url = `${CLOUDMAIL_BASE}/v1/emails?num=${page}&size=10`;
  if (type !== undefined) url += `&type=${type}`;
  const res = await fetch(url, { headers });
  const json = await res.json();
  return json.data;
}

async function getEmail(id) {
  const res = await fetch(`${CLOUDMAIL_BASE}/v1/emails/${id}`, { headers });
  const json = await res.json();
  return json.data;
}

async function deleteEmails(ids) {
  const res = await fetch(`${CLOUDMAIL_BASE}/v1/emails`, {
    method: 'DELETE',
    headers: { ...headers, 'Content-Type': 'application/json' },
    body: JSON.stringify({ emailIds: ids.join(',') }),
  });
  return (await res.json()).code === 200;
}

async function createUser(email, password) {
  const res = await fetch(`${CLOUDMAIL_BASE}/v1/users`, {
    method: 'POST',
    headers: { ...headers, 'Content-Type': 'application/json' },
    body: JSON.stringify({ email, password }),
  });
  return (await res.json()).code === 200;
}

async function listUsers() {
  const res = await fetch(`${CLOUDMAIL_BASE}/v1/users`, { headers });
  const json = await res.json();
  return json.data;
}

async function getStats() {
  const res = await fetch(`${CLOUDMAIL_BASE}/v1/stats`, { headers });
  const json = await res.json();
  return json.data;
}

module.exports = { listEmails, getEmail, deleteEmails, createUser, listUsers, getStats };

Slash Command Examples

📬 /inbox — Cek Email Inbox

const { SlashCommandBuilder, EmbedBuilder } = require('discord.js');
const { listEmails } = require('./api');

module.exports = {
  data: new SlashCommandBuilder()
    .setName('inbox')
    .setDescription('Cek email inbox terbaru')
    .addIntegerOption(opt =>
      opt.setName('page').setDescription('Halaman').setRequired(false)
    ),

  async execute(interaction) {
    const page = interaction.options.getInteger('page') || 1;

    await interaction.deferReply();

    try {
      const data = await listEmails(page, 0);

      if (!data.list.length) {
        return interaction.editReply('📭 Inbox kosong.');
      }

      const embeds = data.list.map(e =>
        new EmbedBuilder()
          .setTitle(e.subject || '(no subject)')
          .setDescription(`From: ${e.from}\n\n${(e.body||'').substring(0,150)}...`)
          .setColor(0x818cf8)
          .setFooter({ text: new Date(e.createTime).toLocaleDateString('id-ID') })
      );

      await interaction.editReply({
        content: `📬 **Page ${page}** — ${data.total} total email`,
        embeds: embeds.slice(0, 5),
      });
    } catch (err) {
      await interaction.editReply(`❌ Error: ${err.message}`);
    }
  },
};

👤 /createuser — Buat User Baru

const { SlashCommandBuilder } = require('discord.js');
const { createUser } = require('./api');

module.exports = {
  data: new SlashCommandBuilder()
    .setName('createuser')
    .setDescription('Buat user CloudMail baru')
    .addStringOption(opt =>
      opt.setName('email').setDescription('Email user').setRequired(true)
    )
    .addStringOption(opt =>
      opt.setName('password').setDescription('Password').setRequired(true)
    ),

  async execute(interaction) {
    const email = interaction.options.getString('email');
    const password = interaction.options.getString('password');

    try {
      const ok = await createUser(email, password);
      await interaction.reply(ok
        ? `✅ User **${email}** berhasil dibuat!`
        : '❌ Gagal membuat user.'
      );
    } catch (err) {
      await interaction.reply(`❌ Error: ${err.message}`);
    }
  },
};

📊 /stats — Statistik Sistem

const { SlashCommandBuilder, EmbedBuilder } = require('discord.js');
const { getStats } = require('./api');

module.exports = {
  data: new SlashCommandBuilder()
    .setName('stats')
    .setDescription('Lihat statistik CloudMail'),

  async execute(interaction) {
    try {
      const stats = await getStats();
      const embed = new EmbedBuilder()
        .setTitle('📊 CloudMail Stats')
        .setColor(0x34d399)
        .addFields(
          { name: '📧 Total Emails', value: `${stats.totalEmails}`, inline: true },
          { name: '👤 Total Users', value: `${stats.totalUsers}`, inline: true },
        );
      await interaction.reply({ embeds: [embed] });
    } catch (err) {
      await interaction.reply(`❌ Error: ${err.message}`);
    }
  },
};
